1. 生成yaml文件
#configtx.yaml
./fabtest -f configtx

#peer,order,zk,kfk yaml
./fabtest -f node

2. 生成 创世区块
./fabtest -c genesisblock

3. 向远程copy crypto channel-artifacts kafkaTLS配置文件
./fabtest -p all

4. 启动节点zk, kfk, order, peer
./fabtest -s zookeeper
./fabtest -s kafka
./fabtest -s order
./fabtest -s peer

5. 创建channel
./fabtest -c channel -channelname testchannel
6. 更新锚节点
./fabtest -r updateanchor -channelname testchannel

7. peer 加入channel
./fabtest -r joinchannel -channelname testchannel

8. 安装chaincode
./fabtest -r installchaincode -ccoutpath ./config/factor.out

9. 实例化chaincode
./fabtest -r runchaincode -ccname factor -channelname testchannel
